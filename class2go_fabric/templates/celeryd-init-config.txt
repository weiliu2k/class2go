# Name of nodes to start, here we have a single node
CELERYD_NODES="{{ app_name }}-default {{ app_name }}-long"

# Where to chdir at start.
CELERYD_CHDIR="<%= @node['system']['admin_home'] %>/{{ app_name }}/main"

# How to call "manage.py celeryd_multi"
CELERYD_MULTI="$CELERYD_CHDIR/manage.py celeryd_multi"

# How to call "manage.py celeryctl"
CELERYCTL="$CELERYD_CHDIR/manage.py celeryctl"

# Extra arguments to celeryd
CELERYD_OPTS="-Q:{{ app_name }}-default {{ app_name }}-default -c:{{ app_name }}-default <%= @node['util']['celery_concurrency'] %> -Q:{{ app_name }}-long {{ app_name }}-long -c:{{ app_name }}-long <%= @node['cpu']['total'] %> --time-limit=<%= @node['util']['celery_timeout'] %>"

# Name of the celery config module.
CELERY_CONFIG_MODULE="celeryconfig"

# %n will be replaced with the nodename.
CELERYD_LOG_FILE="/var/log/celery/%n.log"
CELERYD_PID_FILE="/var/run/celery/%n.pid"

# Workers should run as an unprivileged user.
CELERYD_USER="daemon"
CELERYD_GROUP="daemon"

# Name of the projects settings module.
export DJANGO_SETTINGS_MODULE="settings"
